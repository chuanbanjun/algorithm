#include<bits/stdc++.h>using namespace std;#define mod 1000000007#define maxn 100010int dp[maxn][9][9];int dir[8][2]={{1,2},{-1,2},{1,-2},{-1,-2},{2,1},{2,-1},{-2,1},{-2,-1}};int main(){    dp[0][0][0]=1;    int K,x,y;    scanf("%d%d%d",&K,&x,&y);    for(int t=1;t<=K;t++){        for(int i=0;i<9;i++){            for(int j=0;j<9;j++){                for(int k=0;k<8;k++){                    int nx=i+dir[k][0];                    int ny=j+dir[k][1];                    if(nx>=0&&ny>=0&&nx<9&&ny<9)                        dp[t][i][j]=(dp[t][i][j]+dp[t-1][nx][ny])%mod;                }            }        }    }    printf("%d\n",dp[K][x][y]);    return 0;}
